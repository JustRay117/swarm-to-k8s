# Kubernetes Deployment for web-vote-app
apiVersion: apps/v1
kind: Deployment
metadata:
  name: web-vote-app
spec:
  replicas: 1
  selector:
    matchLabels:
      app: web-vote-app
  template:
    metadata:
      labels:
        app: web-vote-app
    spec:
      containers:
      - name: web-vote-app
        image: ray117/swarm-microservice-demo-v1-web-vote-app:latest
        ports:
        - containerPort: 80
        env:
        - name: WEB_VOTE_NUMBER
          value: "01"
        - name: constraint:node
          value: "=frontend01"

---

# Kubernetes Service for web-vote-app
apiVersion: v1
kind: Service
metadata:
  name: web-vote-app
spec:
  type: NodePort
  ports:
  - port: 8081
    targetPort: 80
  selector:
    app: web-vote-app

---
